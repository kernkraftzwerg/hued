129  void update()
130  {
131  if (_refresh)
132  {
133  _refresh=false;
134  _trefresh.expires_from_now(boost::posix_time::seconds(t_refresh));
135  _trefresh.async_wait(boost::bind(&Responder::refresh, this, placeholders::error));
136 
137  ip::tcp::tcp::resolver resolver(_io_service);
138  ip::tcp::tcp::resolver::query query(_server, _service);
139  ip::tcp::tcp::resolver::iterator endpoint_iterator = resolver.resolve(query);
140 
141  ip::tcp::tcp::socket squery(_io_service);
142  connect(squery, endpoint_iterator);
143 
144  // Build a HTTP request for description.xml
145  streambuf request;
146  std::ostream request_stream(&request);
147  request_stream << "GET /description.xml HTTP/1.0\r\n";
148  request_stream << "Host: " << _server << "\r\n";
149  request_stream << "Accept: */*\r\n";
150  request_stream << "Connection: close\r\n\r\n";
151 
152  // Send the request.
153  write(squery, request);
154 
155  // Read the response status line. The response streambuf will automatically
156  // grow to accommodate the entire line. The growth may be limited by passing
157  // a maximum size to the streambuf constructor.
158  streambuf response;
159  read_until(squery, response, "\r\n");
160 
161  // Check that response is OK.
162  std::istream response_stream(&response);
163  std::string http_version;
164  response_stream >> http_version;
165  uint16_t status_code;
166  response_stream >> status_code;
167  std::string status_message;
168  std::getline(response_stream, status_message);
169 
170  // Invalid response
171  if (!response_stream || http_version.substr(0, 5) != "HTTP/") return;
172 
173  // Wrong status code
174  if (status_code != 200) return;
175 
176  // Read the response headers, which are terminated by a blank line.
177  read_until(squery, response, "\r\n\r\n");
178 
179  // Process the response headers. Just discard the data.
180  std::string header;
181  while (std::getline(response_stream, header) && header != "\r");
182 
183  // Read until EOF
184  boost::system::error_code error;
185  while (read(squery, response,
186  transfer_at_least(1), error));
187  if (error != error::eof)
188  throw boost::system::system_error(error);
189 
190  // Create empty property tree object
191  boost::property_tree::ptree tree;
192 
193  // Parse the XML into the property tree.
194  std::iostream xml(&response);
195  boost::property_tree::read_xml(xml, tree);
196 
197  std::string uuid=tree.get<std::string>("root.device.UDN");
198  const std::string uu("uuid:");
199  if (uuid.find(uu)==0)
200  {
201  _uuid=uuid.substr(uu.length());
202  }
203  }
204  }

218  void operator()(ip::address addr, uint16_t port, uint16_t mx)

222  _tresponse.expires_from_now(boost::posix_time::milliseconds(t_response));
223  _tresponse.async_wait(boost::bind(&Responder::respond, this, placeholders::error, addr, port));

245  void respond(const boost::system::error_code &e, ip::address addr, uint16_t port)
246  {
247  if (e) return;
248 
249  const ip::udp::endpoint endpoint(addr, port);
250 
251  boost::format st1(HUE_ST1);
252  st1%_uuid;
253  boost::format st2(HUE_ST2);
254  st2%_uuid;
255  boost::format st3(HUE_ST3);
256  st3%_uuid;
257  boost::format msg1(HUE_RESPONSE);
258  boost::format msg2(HUE_RESPONSE);
259  boost::format msg3(HUE_RESPONSE);
260  msg1%_server%_service%_uuid%st1;
261  msg2%_server%_service%_uuid%st2;
262  msg3%_server%_service%_uuid%st3;
263  _socket.send_to(buffer(msg1.str()), endpoint);
264  _socket.send_to(buffer(msg2.str()), endpoint);
265  _socket.send_to(buffer(msg3.str()), endpoint);
266  }

271 class Listener
272 {
273 private:
274  Responder &_resp;

277  static const uint16_t _max_length=1024;

280 public:
284  Listener(io_service &io_service, Responder &resp, const ip::address &listen_address,
285  const ip::address &multicast_address) :
286  _resp(resp), _socket(io_service)
287  {
288  // Create the socket so that multiple may be bound to the same address.
289  ip::udp::endpoint listen_endpoint(listen_address, multicast_port);
290  _socket.open(listen_endpoint.protocol());
291  _socket.set_option(ip::udp::socket::reuse_address(true));
292  _socket.bind(listen_endpoint);
293 
294  // Join the multicast group.
295  _socket.set_option(ip::multicast::join_group(multicast_address));
296 
297  _socket.async_receive_from(buffer(_data, _max_length), _sender_endpoint,
298  boost::bind(&Listener::receive, this, placeholders::error,
299  placeholders::bytes_transferred));
300  }
301 
313  void receive(const boost::system::error_code &error, size_t bytes)
314  {
315  if (error) return;
316 
317  _socket.async_receive_from(buffer(_data, _max_length), _sender_endpoint,
318  boost::bind(&Listener::receive, this, placeholders::error,
319  placeholders::bytes_transferred));
320 
321  const std::regex msearch("^M-SEARCH \\* HTTP/1\\.1");
322  const std::regex expr{"(\\S+):\\s(\\S+)"};
323  std::match_results<char*> what;
324 
325  if (std::regex_search(_data, _data+bytes, what, msearch))
326  {
327  std::unordered_map<std::string, std::string> request;
328  for (char* start=what[0].second;
329  regex_search(start, _data+bytes, what, expr);
330  start=what[2].second) request[what[1]]=what[2];
331 
332  // Is this a supported service type?
333  if (service_types.find(request["ST"])==service_types.end()) return;
334 
335  try
336  {
337  _resp(_sender_endpoint.address(), _sender_endpoint.port(), boost::lexical_cast<uint16_t>(request["MX"]));
338  }
339  catch(const boost::bad_lexical_cast &)
340  {
341  }
342  }
343  }
